# 递归

```
var combine = function (n, k) {
    //start:起始位置，path记录组合
    const backtracking = function (n, k, start, path) {
        if (path.length === k) {
            //深拷贝数组
            res.push([...path]);
						//res.push(Array.from(path));
            return;
        }
        for (let i = start; i <= n - (k - path.length) + 1; i++) {
            path.push(i);
            backtracking(n, k, i + 1, path);
            path.pop();
        }
    }
    let res = [];
    backtracking(n, k, 1, []);
    return res;
};
```
