> Problem: [78. 子集](https://leetcode.cn/problems/subsets/description/)

# 思路
本题使用回溯法，不断取数组的字集。

# 代码
```JavaScript []

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var subsets = function (nums) {
    const res = [], path = [];
    //回溯主体
    var backtraking = function (start) {
        res.push([...path]);
        for (let i = start; i < nums.length; i++) {
            path.push(nums[i]);
            backtraking(i + 1);
            path.pop();
        }
    }
    backtraking(0);
    return res;
};
